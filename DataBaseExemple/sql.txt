CREATE TABLE City (
    id SERIAL PRIMARY KEY, 
    name VARCHAR(100) UNIQUE NOT NULL,
    region VARCHAR(100) NOT NULL, 
    district VARCHAR(100) NOT NULL,
    population INTEGER NOT NULL,
    foundation INTEGER NOT NULL
);
